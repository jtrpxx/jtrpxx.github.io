from . import views // import ไฟล์ views.py ภายใน directory เดียวกัน เพื่อเรียกใช้ฟังก์ชันที่อยู่ใน view
app_name = 'contract_collateral' // กำหนด namespace เพื่อแยกชื่อ path ของแต่ละแอป

from django.db import models // import models
from utils.function import BaseModel // import class BaseModel

class CollateralInsure(BaseModel): //สืบทอด class BaseModel
    collateral = models.ForeignKey(CollateralInfo, related_name='collateral_insure_collateral', on_delete=models.CASCADE, verbose_name='หลักทรัพย์')

    insure_no = models.CharField(max_length=50, null=True, blank=True, verbose_name='เลขที่กรมธรรม์')

    class Meta:

        ordering = ['id']

from insurance.collateral_insure.models import CollateralInsure // import CollateralInsure
from contract.hirepurchase.models import ContractInfo // import ContractInfo
 
def detail(request, slug): // เก็บข้อมูลทั้งหมดที่ส่งมาจาก client 
    collateral_info = get_object_or_404(CollateralInfo, slug=slug) // ดึงข้อมูลจาก Model CollateralInfo ที่มี slug ตรงกับค่าที่ส่งมา 
    collateral_insure = CollateralInsure.objects.filter( // ดึงข้อมูลจาก Model CollateralInsure ที่สัมพันธ์กับ collateral_info
        collateral=collateral_info,
        status='A'
    ).order_by('insure_date', 'insure_expire', 'id') 

    context = { //  สร้างชุดข้อมูล ที่จะส่งไปให้ Template
        'collateral_info': collateral_info,
        'collateral_insure': collateral_insure,
    }

    return render(request, 'collateral/detail.html', context) // ใช้ฟังก์ชัน render() เพื่อประมวลผล Template collateral/detail.html และส่ง context ไปให้ template เพื่อแสดงข้อมูลจริงในหน้าเว็บ

ในตัวอย่างนี้ ผมแสดงให้เห็นการออกแบบและพัฒนาโปรเจกต์ด้วยแนวคิด MVC โดยแยกหน้าที่ของแต่ละส่วนอย่างชัดเจน เช่น Model สำหรับจัดการข้อมูล, View สำหรับแสดงผล และ Controller สำหรับควบคุมการทำงานของระบบ ช่วยให้โครงสร้างโค้ดสามารถดูแลรักษาได้ง่าย รองรับการทำงานร่วมกันเป็นทีม และสามารถนำไปปรับปรุงหรือขยายต่อในอนาคตได้อย่างยืดหยุ่น อีกทั้งยังมีการใช้แนวคิด OOP เช่น การสร้าง Model ให้สืบทอดจาก BaseModel เพื่อให้สามารถใช้ฟังก์ชันร่วมกันได้

ในด้านการออกแบบฐานข้อมูล ผมได้จัดทำ Data Dictionary อย่างเป็นระบบ โดยใส่ความหมายของแต่ละฟิลด์ผ่านคอลัมน์ Comment และในแต่ละฟิลด์ก็ตั้งชื่อให้สื่อความหมายเข้าใจง่าย เพื่อให้นักพัฒนาคนอื่นสามารถเข้าใจหน้าที่ของฟิลด์ได้ทันทีโดยไม่ต้องเดา นอกจากนี้ยังมีการกำหนด Foreign Key เพื่อเชื่อมโยงข้อมูลระหว่างตารางที่เกี่ยวข้องกัน และใช้ Index เพื่อเพิ่มประสิทธิภาพในการค้นหาข้อมูล ช่วยให้ระบบทำงานได้อย่างรวดเร็วและเสถียร โดยเฉพาะเมื่อต้องจัดการข้อมูลจำนวนมากหรือทำงานร่วมกับหลายตาราง
